import ReactDOM from 'react-dom/client'
var date=new Date()
var day=Date.getday()
let result;
switch(day){
  case 0:
    result="sunday";
    break;
      case 1:
    result="Monday";
    break;
      case 2:
    result="Tuesday";
    break;
      case 3:
    result="Wednesday";
    break;
      case 4:
    result="Thursday";
    break;
      case 5:
    result="Friday";
    break;
  case 6:
    result="Saturday";
    default:
      result:"day not matched"
}

 const root=ReactDOM.createRoot(document.getElementById('root'));
root.render(
  <>
<h1 style={{ fontStyle:"italic" ,color:"green"}}>{result}</h1>
</>
);
